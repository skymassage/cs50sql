Table customers {
  id INTEGER [primary key]
  username TEXT
  gender TEXT
  age TEXT
  city TEXT
}

Table items {
  id INT [primary key]
  title TEXT
  category_id INTEGER [ref: > categories.id]
  price NUMERIC 
  number INTEGER
  creating_time NUMBERIC
  active INTEGER
}

Table categories {
  id INTEGER [primary key]
  name TEXT
}

TABLE orders {
  id INTEGER [primary key]
  customer_id INTEGER [ref: > customers.id]
  creating_time NUMERIC
}

TABLE order_details {
  order_id INTEGER [ref: > orders.id]
  item_id INTEGER [ref: > items.id]
  price NUMERIC
  number INTEGER

  indexes {
    (order_id, item_id) [pk] 
  }
}

TABLE comments {
  id INTEGER [primary key]
  customer_id INTEGER [ref: > customers.id]
  item_id INTEGER [ref: > items.id]
  content TEXT
  creating_time NUMERIC
}

TABLE rates {
  customer_id INRGER [ref: > customers.id]
  item_id INTEGER [ref: > items.id]
  rating INTEGER

  indexes {
    (customer_id, item_id) [pk]
  }
}

Table watchlists {
  customer_id INTEGER [ref: > customers.id]
  item_id INTEGER [ref: > items.id]

  indexes {
    (customer_id, item_id) [pk] 
  }
}
